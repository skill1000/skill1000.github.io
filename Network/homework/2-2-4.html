undo terminal monitor

system-view
sysname FW1_QiaoXiang

ospf 1

area 0
network 172.16.20.0 0.0.0.255
network 172.16.30.0 0.0.0.255
quit

quit

undo firewall packet-filter basic-protocol enable

interface Eth-Trunk0
ip address 10.10.10.1 255.255.255.0
quit

interface GigabitEthernet1/0/1
ip address 10.5.1.2 30
ospf enable 1 area 0
ospf cost 10
quit

interface GigabitEthernet1/0/2
ip address 10.3.1.2 30
ospf enable 1 area 0
quit

interface GigabitEthernet1/0/3
eth-trunk 0
quit

interface GigabitEthernet1/0/4
eth-trunk 0
quit

interface GigabitEthernet1/0/5
ip address 172.16.30.2 24
vrrp vrid 2 virtual-ip 172.16.30.1 active
ospf enable 1 area 0
quit

interface GigabitEthernet1/0/6
ip address 172.16.20.2 24
vrrp vrid 1 virtual-ip 172.16.20.1 active
ospf enable 1 area 0
quit

firewall zone untrust
add interface GigabitEthernet 1/0/1
add interface GigabitEthernet 1/0/2
quit

firewall zone dmz
add interface Eth-Trunk0
quit

firewall zone trust
add interface GigabitEthernet 1/0/5
add interface GigabitEthernet 1/0/6
quit

hrp track interface GigabitEthernet1/0/1
hrp track interface GigabitEthernet1/0/3
hrp adjust ospf-cost enable
hrp mirror session enable
hrp interface Eth-Trunk0 remote 10.10.10.2
hrp enable

security-policy

rule name trust-untrust_QiaoXiang
source-zone trust
destination-zone untrust
action permit
rule name untrust-trust
source-zone untrust
destination-zone trust
action permit
quit

quit
