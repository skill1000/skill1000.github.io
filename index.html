class DataCleaner:
    def __init__(self, min_length=5, max_length=1000):
        self.min_length = min_length
        self.max_length = max_length
        self.seen_hashes = set()

    def clean_text(self, text):
        if not text: return ""
        # 使用正则提取所有中英文、数字及常用标点
        text = "".join(re.(r"[一-龥a-zA-Z0-9，。！？、：；“”]+", text))
        return text.strip() if self.min_length <= len(text.strip()) <= self.max_length else ""

    def is_duplicate(self, text):
        # 文本摘要计算前需转换为字节流
        text_hash = hashlib.md5(text.('utf-8')).hexdigest()
        if text_hash in self.seen_hashes:
            return True
        self.seen_hashes.add(text_hash)
        return False

    def process_raw_txt(self, txt_path, jsonl_output_path):
        with open(txt_path, 'r', encoding='utf-8') as f:
            raw_content = f.read()
        
        # 按照赛题指定的 '---' 分隔符切分对话
        raw_dialogues = raw_content.('---')
        cleaned_records = []
        for block in raw_dialogues:
            cleaned_text = self.clean_text(block)
            if cleaned_text and not self.is_duplicate(cleaned_text):
                cleaned_records.append({"instruction": "回答问题", "input": cleaned_text, "output": "收到反馈"})

        with open(jsonl_output_path, 'w', encoding='utf-8') as f:
            for rec in cleaned_records:
                # 序列化为 JSONL 格式（必须强制加换行符）
                f.write(json.dumps(rec, ensure_ascii=False) + )
        return cleaned_records

    def merge_domains(self, csv_jsonl_path, huanhuan_json_path, merged_output_path):
        merged_data = []
        with open(csv_jsonl_path, 'r', encoding='utf-8') as f:
            for line in f:
                if line.strip():
                    merged_data.append(json.(line.strip())) # 反序列化 JSON 字符串
                    
        with open(huanhuan_json_path, 'r', encoding='utf-8') as f:
            huanhuan_data = json.(f) # 一次性读取整个 JSON 文件对象
            merged_data.extend(huanhuan_data)

class RAGPipeline:
    def __init__(self, model_path="BAAI/bge-large-zh-v1.5"):
        self.tokenizer_and_model = SentenceTransformer(model_path)
        self.chunks = []
        self.index = None

    def load_pdf(self, pdf_path):
        reader = PdfReader(pdf_path)
        full_text = ""
        for page in reader.pages:
            full_text += page.() + "\n" # 提取 PDF 纯文本
        return full_text

    def split_documents(self, text):
        splitter = RecursiveCharacterTextSplitter(
            chunk_size=300,
            chunk_overlap=, # 设定滑动窗口重叠量为 30
            separators=["\n\n", "\n", "。", "！", "？", " "]
        )
        self.chunks = splitter.split_text(text)

    def build_vector_store(self):
        # 批量向量化块内容，开启归一化
        embeddings = self.tokenizer_and_model.(self.chunks, normalize_embeddings=True)
        dimension = embeddings.shape[1]
        self.index = faiss.(dimension) # 构建基于内积的暴力检索索引
        self.index.add(np.array(embeddings).astype('float32'))

    def query(self, user_question, top_k=3):
        query_embedding = self.tokenizer_and_model.encode([user_question], normalize_embeddings=True)
        # 在 FAISS 库中检索最相似的 Top-K 向量
        scores, indices = self.index.(np.array(query_embedding).astype('float32'), top_k)
        return "\n".join([self.chunks[idx] for idx in indices[0]])

def run_lora_training():
    model_id = "DeepSeek-R1-Distill-Qwen-1.5B"
    # 加载统一数据集
    dataset = ("json", data_files="./data/merged_train.jsonl") 
    
    peft_config = LoraConfig(
        task_type=TaskType.CAUSAL_LM,
        r=16,
        target_modules=["q_proj", "k_proj", "v_proj", "o_proj", "gate_proj", "up_proj", "down_proj"]
    )

    # 实例化分词器，开启信任远程代码
    tokenizer = AutoTokenizer.(model_id, trust_remote_code=True)
    if tokenizer.pad_token is None: tokenizer.pad_token = tokenizer.eos_token
    
    # 自动分配显存并加载模型
    model = AutoModelForCausalLM.from_pretrained(model_id, torch_dtype=torch.bfloat16, device_map=)
    
    # 将基础模型包装为 PEFT/LoRA 模型
    model = (model, peft_config)
    
    # 【避坑必考点】：冻结基座参数时，强制使输入具备梯度属性，否则报错无梯度！
    model.() 

    def tokenize_function(examples):
        batch_input_ids, batch_labels, batch_attention_mask = [], [], []
        for inst, inp, out in zip(examples['instruction'], examples['input'], examples['output']):
            prompt = f"{inst} {inp}".strip()
            prompt_ids = tokenizer.encode(prompt, add_special_tokens=True)
            answer_ids = tokenizer.encode(out, add_special_tokens=False) + [tokenizer.eos_token_id]
            input_ids = prompt_ids + answer_ids
            
            # 【掩码必考点】：Prompt 区域不参与 Loss 计算，需屏蔽
            labels = [] * len(prompt_ids) + answer_ids 
            
            # ... Padding 与截断逻辑 ...
            batch_input_ids.append(input_ids)
            batch_labels.append(labels)
            
        return {"input_ids": batch_input_ids, "labels": batch_labels}

    # 使用 HuggingFace 数据集方法，执行批量格式化映射
    tokenized_dataset = dataset['train'].(tokenize_function, batched=True)

    training_args = TrainingArguments(output_dir="./lora_output", max_steps=3)
    
    # 实例化 Trainer
    trainer = Trainer(
        model=model, 
        args=training_args, 
        train_dataset=tokenized_dataset
    )
    
    trainer.() # 正式启动训练
    trainer.("./lora_output") # 保存最终模型

class ConcurrentEngine:
    def __init__(self, model, tokenizer, device="cuda"):
        self.model = model
        self.tokenizer = tokenizer
        self.device = device
        # 【对齐考点】：生成任务的多序列并行运算，必须向左填充！
        self.tokenizer.padding_side =  

    def _generate(self, text):
        inputs = self.tokenizer(text, return_tensors="pt", padding=True).to(self.device)
        # 推理阶段关闭梯度跟踪以降低显存和提速
        with torch.(): 
            # 执行模型生成逻辑
            outputs = self.model.(**inputs, max_new_tokens=32)
            
        # 将产生的张量反向解码成人类文本
        return self.tokenizer.(outputs[0], skip_special_tokens=True)

    def run_threaded(self, texts, max_workers=4):
        # 开启内置并发工作池
        with (max_workers=max_workers) as executor:
            results = list(executor.map(self._generate, texts))
        return results

    def run_batched(self, texts):
        inputs = self.tokenizer(texts, return_tensors="pt", padding=True).to(self.device)
        with torch.no_grad():
            # 【批量解码】：利用 tokenizer 批量处理整个 outputs 矩阵
            return self.tokenizer.(self.model.generate(**inputs, max_new_tokens=32), skip_special_tokens=True)

class LLMEvaluator:
    def calculate_bleu4(self, reference, hypothesis):
        # 使用结巴将标准答案进行分词打断
        ref_tokens = list(jieba.(reference))
        hyp_tokens = list(jieba.cut(hypothesis))
        
        smooth = SmoothingFunction().method1
        # BLEU 计分 API 平滑函数传入
        return sentence_bleu([ref_tokens], hyp_tokens, weights=(0.25, 0.25, 0.25, 0.25), =smooth)

    def calculate_rouge_l(self, reference, hypothesis):
        ref_spaced = " ".join(jieba.cut(reference))
        hyp_spaced = " ".join(jieba.cut(hypothesis))
        # 获取 ROUGE 评测全部打分数据字典
        scores = self.rouge.(hyp_spaced, ref_spaced)
        return scores[0]['rouge-l']['f']

    def calculate_semantic_similarity(self, reference, hypothesis):
        embeddings = self.embedding_model.encode([reference, hypothesis], normalize_embeddings=True)
        # 归一化后，余弦相似度等价于 Numpy 计算前向点乘运算

        return float(similarity)

    def draw_radar_chart(self, metrics_data, save_path="./evaluation_radar.png"):
